/*
 * =============================================================
 *  Программа: Класс Student
 *  Описание:  Демонстрация ООП на C++ — создание класса Student
 *             с инкапсуляцией, сортировкой и фильтрацией данных.
 *  Автор:     Учебный пример
 *  Компиляция: g++ -std=c++17 -o student main.cpp
 * =============================================================
 */

#include <iostream>   // Ввод/вывод
#include <string>     // Работа со строками
#include <algorithm>  // std::sort, std::all_of
#include <iomanip>    // std::fixed, std::setprecision

using namespace std;

// ============================================================
// Класс Student — представляет студента с оценками
// ============================================================
class Student {
private:
    string fullName;      // Фамилия и инициалы (например, "Иванов И.И.")
    string groupNumber;   // Номер группы (например, "ИС-21")
    int grades[5];        // Статический массив из 5 оценок (от 1 до 5)

public:
    // ----------------------------------------------------------
    // Конструктор по умолчанию
    // ----------------------------------------------------------
    Student() : fullName(""), groupNumber("") {
        for (int i = 0; i < 5; i++)
            grades[i] = 0;
    }

    // ----------------------------------------------------------
    // Конструктор с параметрами: имя, группа и массив оценок
    // ----------------------------------------------------------
    Student(const string& name, const string& group, const int gr[5])
        : fullName(name), groupNumber(group)
    {
        for (int i = 0; i < 5; i++)
            grades[i] = gr[i];
    }

    // ----------------------------------------------------------
    // Конструктор с параметрами: имя, группа и 5 отдельных оценок
    // ----------------------------------------------------------
    Student(const string& name, const string& group,
            int g1, int g2, int g3, int g4, int g5)
        : fullName(name), groupNumber(group)
    {
        grades[0] = g1;
        grades[1] = g2;
        grades[2] = g3;
        grades[3] = g4;
        grades[4] = g5;
    }

    // ----------------------------------------------------------
    // Метод: вычисление среднего балла студента
    // Возвращает среднее арифметическое пяти оценок как double
    // ----------------------------------------------------------
    double getAverage() const {
        int sum = 0;
        for (int i = 0; i < 5; i++)
            sum += grades[i];
        return static_cast<double>(sum) / 5.0;
    }

    // ----------------------------------------------------------
    // Метод: проверка, все ли оценки равны 4 или 5
    // Возвращает true, если нет ни одной оценки ниже 4
    // ----------------------------------------------------------
    bool hasOnlyGoodGrades() const {
        for (int i = 0; i < 5; i++) {
            if (grades[i] < 4)
                return false;
        }
        return true;
    }

    // ----------------------------------------------------------
    // Геттеры для доступа к закрытым полям
    // ----------------------------------------------------------
    string getFullName() const {
        return fullName;
    }

    string getGroupNumber() const {
        return groupNumber;
    }

    // ----------------------------------------------------------
    // Метод: вывод полной информации о студенте
    // ----------------------------------------------------------
    void printInfo() const {
        cout << fullName
             << ", группа " << groupNumber
             << ", оценки: ";
        for (int i = 0; i < 5; i++) {
            cout << grades[i];
            if (i < 4) cout << " ";
        }
        cout << ", средний балл: "
             << fixed << setprecision(1)
             << getAverage() << endl;
    }
};


// ============================================================
//  Функция main — точка входа в программу
// ============================================================
int main() {
    // Установка кодировки для корректного вывода кириллицы (Windows)
    // system("chcp 65001");  // раскомментировать при необходимости

    const int N = 10;  // Количество студентов

    // ----------------------------------------------------------
    // Вариант 1: Ввод данных с клавиатуры (раскомментировать)
    // ----------------------------------------------------------
    /*
    Student students[N];
    for (int i = 0; i < N; i++) {
        string name, group;
        int g[5];

        cout << "Студент #" << (i + 1) << ":" << endl;
        cout << "  Фамилия и инициалы: ";
        getline(cin, name);
        cout << "  Номер группы: ";
        getline(cin, group);
        cout << "  Введите 5 оценок через пробел: ";
        for (int j = 0; j < 5; j++)
            cin >> g[j];
        cin.ignore();  // Очистка буфера после cin >>

        students[i] = Student(name, group, g);
    }
    */

    // ----------------------------------------------------------
    // Вариант 2: Данные заданы жёстко (для демонстрации)
    // ----------------------------------------------------------
    Student students[N] = {
        Student("Иванов И.И.",    "ИС-21", 5, 4, 5, 5, 4),
        Student("Петров П.П.",    "ИС-21", 3, 3, 4, 2, 3),
        Student("Сидоров С.С.",   "ИС-22", 5, 5, 4, 5, 5),
        Student("Козлова А.В.",   "ИС-22", 4, 3, 4, 4, 3),
        Student("Новикова Е.А.",  "ИС-23", 4, 4, 5, 4, 4),
        Student("Морозов Д.К.",   "ИС-21", 3, 3, 3, 3, 2),
        Student("Волкова М.С.",   "ИС-23", 5, 5, 5, 5, 5),
        Student("Лебедев Р.О.",   "ИС-22", 4, 4, 4, 5, 4),
        Student("Соколов В.Н.",   "ИС-23", 3, 4, 3, 4, 3),
        Student("Кузнецова Т.И.", "ИС-21", 5, 4, 4, 5, 5)
    };

    // ----------------------------------------------------------
    // Сортировка массива по возрастанию среднего балла
    // Используем std::sort с лямбда-компаратором
    // ----------------------------------------------------------
    sort(students, students + N, [](const Student& a, const Student& b) {
        return a.getAverage() < b.getAverage();
    });

    // ----------------------------------------------------------
    // Вывод отсортированного списка всех студентов
    // ----------------------------------------------------------
    cout << "=========================================" << endl;
    cout << " Отсортированный список студентов" << endl;
    cout << " (по возрастанию среднего балла)" << endl;
    cout << "=========================================" << endl;

    for (int i = 0; i < N; i++) {
        cout << setw(2) << (i + 1) << ". "
             << students[i].getFullName()
             << ", группа " << students[i].getGroupNumber()
             << ", средний балл: "
             << fixed << setprecision(1)
             << students[i].getAverage()
             << endl;
    }

    // ----------------------------------------------------------
    // Вывод студентов, имеющих только оценки 4 и 5
    // ----------------------------------------------------------
    cout << endl;
    cout << "=========================================" << endl;
    cout << " Студенты с оценками только 4 и 5" << endl;
    cout << "=========================================" << endl;

    bool found = false;  // Флаг: найден ли хотя бы один такой студент

    for (int i = 0; i < N; i++) {
        if (students[i].hasOnlyGoodGrades()) {
            cout << "  " << students[i].getFullName()
                 << ", группа " << students[i].getGroupNumber()
                 << endl;
            found = true;
        }
    }

    // Если ни одного студента с оценками 4 и 5 не найдено
    if (!found) {
        cout << "  Нет студентов с оценками только 4 и 5." << endl;
    }

    cout << "=========================================" << endl;

    return 0;
}
